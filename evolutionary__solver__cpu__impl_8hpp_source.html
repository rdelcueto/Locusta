100template<typename TFloat>
101void
102evolutionary_solver_cpu<TFloat>::update_records()
103{
104 TFloat* genomes = _population->_data_array;
105 TFloat* fitness = _population->_fitness_array;
106
107 // Scan population
108 for (uint32_t i = 0; i < _ISLES; i++) {
109
110 uint32_t isle_max_idx = 0;
111 TFloat isle_max_fitness = fitness[i * _AGENTS];
112 uint32_t isle_min_idx = 0;
113 TFloat isle_min_fitness = fitness[i * _AGENTS];
114
115 for (uint32_t j = 1; j < _AGENTS; j++) {
116
117 const TFloat candidate_fitness = fitness[i * _AGENTS + j];
118
119 if (candidate_fitness > isle_max_fitness) {
120 isle_max_fitness = candidate_fitness;
121 isle_max_idx = j;
122 }
123
124 if (candidate_fitness < isle_min_fitness) {
125 isle_min_fitness = candidate_fitness;
126 isle_min_idx = j;
127 }
128 }
129
130 // Update current isle records
131 const TFloat curr_max = _max_agent_fitness[i];
132
133 if (isle_max_fitness > curr_max) {
134 _max_agent_idx[i] = isle_max_idx;
135 _max_agent_fitness[i] = isle_max_fitness;
136
137 TFloat* max_genome = _max_agent_genome + i * _DIMENSIONS;
138 const uint32_t max_genomes_offset =
139 i * _AGENTS * _DIMENSIONS + isle_max_idx * _DIMENSIONS;
140
141 for (uint32_t k = 0; k < _DIMENSIONS; k++) {
142 max_genome[k] = genomes[max_genomes_offset + k];
143 }
144 }
145
146 const TFloat curr_min = _min_agent_fitness[i];
147
148 if (isle_min_fitness < curr_min) {
149 _min_agent_idx[i] = isle_min_idx;
150 _min_agent_fitness[i] = isle_min_fitness;
151
152 TFloat* min_genome = _min_agent_genome + i * _DIMENSIONS;
153 const uint32_t min_genomes_offset =
154 i * _AGENTS * _DIMENSIONS + isle_min_idx * _DIMENSIONS;
155
156 for (uint32_t k = 0; k < _DIMENSIONS; k++) {
157 min_genome[k] = genomes[min_genomes_offset + k];
