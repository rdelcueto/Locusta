209 // Scan population

211 for (uint32_t j = 0; j < _AGENTS; j++) {
212 const uint32_t FITNESS_OFFSET = i * _AGENTS + j;
213 if (trial_fitness[FITNESS_OFFSET] > previous_fitness[FITNESS_OFFSET]) {
214 // Replace target vector with trial vector
