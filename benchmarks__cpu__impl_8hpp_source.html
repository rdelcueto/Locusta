68template<typename TFloat>
69inline TFloat
70osz(const TFloat x, const uint32_t i, const uint32_t k)
71{
72 const TFloat c3 = 0.049;
73
74 if (i == 0 || i == (k - 1)) {
75
76 if (x == 0) {
77 return 0;
78 } else {
79 const TFloat xx = log(fabs(x));
80 if (x > 0) {
81 const TFloat c1 = 10;
82 const TFloat c2 = 7.9;
83
84 return exp(xx + c3 * (sin(c1 * xx) + sin(c2 * xx)));
85 } else {
86 const TFloat c1 = 5.5;
87 const TFloat c2 = 3.1;
88
89 return -exp(xx + c3 * (sin(c1 * xx) + sin(c2 * xx)));
90 }
91 }
92
93 } else {
94 return x;
95 }

323template<typename TFloat>
324inline TFloat
325ackley(const uint32_t DIMENSIONS,
326 const uint32_t DIMENSION_OFFSET,
327 const TFloat* evaluation_vector)
328{
329 const TFloat PI = 3.1415926535;
330 const TFloat E = 2.7182818284;
331 const TFloat beta = 0.5;
332 const TFloat c1 = 10.0;
333 const TFloat c2 = 1.0;
334 const TFloat c3 = 2.0;
335 const TFloat c4 = 0.2;
336 const TFloat c5 = 20.0;
337
338 TFloat reduction_sum_a = 0.0;
339 TFloat reduction_sum_b = 0.0;
340
341#pragma omp simd reduction(+ : reduction_sum_a, reduction_sum_b)
342 for (uint32_t k = 0; k < DIMENSIONS; ++k) {
343 const TFloat x = evaluation_vector[k * DIMENSION_OFFSET];
344 const TFloat asy_x = asy<TFloat>(x, beta, k, DIMENSIONS);
345 const TFloat y = asy_x * pow(c1, c2 * k / (DIMENSIONS - 1) / c3);
346
347 reduction_sum_a += y * y;
348 reduction_sum_b += cos(c3 * PI * y);
349 }
350
351 reduction_sum_a = -c4 * sqrt(reduction_sum_a / DIMENSIONS);
352 reduction_sum_b /= DIMENSIONS;
353
354 return E - c5 * exp(reduction_sum_a) - exp(reduction_sum_b) + c5;
355}
364template<typename TFloat>
365inline TFloat
366weierstrass(const uint32_t DIMENSIONS,
367 const uint32_t DIMENSION_OFFSET,
368 const TFloat* evaluation_vector)
369{
370 const TFloat PI = 3.14159;
371 const TFloat beta = 0.5;
372 const TFloat c1 = 10.0;
373 const TFloat c2 = 1.0;
374 const TFloat c3 = 2.0;
375 const TFloat c4 = 0.5;
376
377 const TFloat a = 0.5;
378 const TFloat b = 3.0;
379
380 const uint32_t it_max = 20;
381
382 TFloat reduction_sum_a = 0.0;
383 TFloat reduction_sum_b;
384 TFloat reduction_sum_c;
385
386 for (uint32_t k = 0; k < DIMENSIONS; ++k) {
387 reduction_sum_b = 0.0;
388 reduction_sum_c = 0.0;
389
390 const TFloat x = evaluation_vector[k * DIMENSION_OFFSET];
391 const TFloat asy_x = asy<TFloat>(x, beta, k, DIMENSIONS);
392 const TFloat y = asy_x * pow(c1, c2 * k / (DIMENSIONS - 1) / c3);
393
394 TFloat pow_a = 1;
395 TFloat pow_b = 1;
396
397#pragma omp simd reduction(+ : reduction_sum_b, reduction_sum_c) \
398 reduction(* : pow_a, pow_b)
399 for (uint32_t it = 0; it <= it_max; ++it) {
400 reduction_sum_b += pow_a * cos(c3 * PI * pow_b * (y + c4));
401 reduction_sum_c += pow_a * cos(c3 * PI * pow_b * c4);
402 pow_a *= a;
403 pow_b *= b;
404 }
405
406 reduction_sum_a += reduction_sum_b - DIMENSIONS * reduction_sum_c;
407 }
408 return reduction_sum_a;
409}

421griewank(const uint32_t DIMENSIONS,

427 const TFloat c3 = 2.0;
428 const TFloat c4 = 1.0;
429 const TFloat c5 = 4000.0;
430
431 TFloat reduction_sum = 0.0;
432 TFloat reduction_prod = 1.0;
433
434#pragma omp simd reduction(+ : reduction_sum) reduction(* : reduction_prod)
435 for (uint32_t k = 0; k < DIMENSIONS; ++k) {
436 const TFloat x = evaluation_vector[k * DIMENSION_OFFSET];
437 const TFloat y = x * pow(c1, c2 * k / (DIMENSIONS - 1) / c3);
438 reduction_sum += y * y;
439 reduction_prod *= cos(y / sqrt(c2 + k));
440 }
441 return c4 + reduction_sum / c5 - reduction_prod;

503template<typename TFloat>
504void
505benchmark_dispatch(const uint32_t ISLES,
506 const uint32_t AGENTS,
507 const uint32_t DIMENSIONS,
508 const bool F_NEGATE_EVALUATION,
509 const uint32_t FUNC_ID,
510 const TFloat EVALUATION_BIAS,
511 const TFloat* SHIFT_ORIGIN,
512 const bool F_ROTATE,
513 const TFloat* ROTATION_MATRIX,
514 const TFloat* evaluation_data,
515 TFloat* evaluation_results,
516 prngenerator_cpu<TFloat>* local_generator)
517{
518
519 const uint32_t REPETITIONS = 1e0;
520 TFloat (*benchmark_function)(const uint32_t, const uint32_t, const TFloat*);
521
522 switch (FUNC_ID) {
523
524 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_ELLIPS:
525 benchmark_function = ellips;
526 break;
527 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_BENT_CIGAR:
528 benchmark_function = bent_cigar;
529 break;
530 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_DISCUS:
531 benchmark_function = discus;
532 break;
533 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::DIFF_POWERS:
534 benchmark_function = diff_powers;
535 break;
536 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_ROSENBROCK:
537 benchmark_function = rosenbrock;
538 break;
539 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_SCHAFFER:
540 benchmark_function = schaffer;
541 break;
542 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_ACKLEY:
543 benchmark_function = ackley;
544 break;
545 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_WEIERSTRASS:
546 benchmark_function = weierstrass;
547 break;
548 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_GRIEWANK:
549 benchmark_function = griewank;
550 break;
551 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::RASTRIGIN:
552 benchmark_function = rastrigin;
553 break;
554 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::ROT_RASTRIGIN:
555 benchmark_function = rastrigin;
556 break;
557 case BenchmarkFunctor<TFloat>::FUNCTION_IDENTIFIERS::SPHERE:
558 default:
559 benchmark_function = sphere;
560 break;
561 }
562
563#pragma omp parallel for collapse(2)
564 for (uint32_t i = 0; i < ISLES; ++i) {
565 for (uint32_t j = 0; j < AGENTS; ++j) {
566 const uint32_t isle = i;
567 const uint32_t agent = j;
568 const uint32_t OFFSET = 1;
569
570 TFloat result = 0;
571 const TFloat* genome =
572 evaluation_data + i * AGENTS * DIMENSIONS + j * DIMENSIONS;
573
574 for (uint32_t r = 0; r < REPETITIONS; ++r) {
575 result = benchmark_function(DIMENSIONS, OFFSET, genome);
576 }
577
578 result += EVALUATION_BIAS;
579
580 evaluation_results[isle * AGENTS + agent] =
581 F_NEGATE_EVALUATION ? -result : result;
582 }
583 }
584}
585}
