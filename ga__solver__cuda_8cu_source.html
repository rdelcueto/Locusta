8/**
9 * @brief CUDA kernel for replacing the elite population.
10 *
11 * This kernel replaces the elite population with the best individuals from the
12 * current population.
13 *
14 * @param AGENTS Number of agents per isle.
15 * @param DIMENSIONS Number of dimensions per agent.
16 * @param min_agent_idx Device array of indices of the agents with minimum
17 * fitness.
18 * @param max_agent_genome Device array of genomes with maximum fitness.
19 * @param max_agent_fitness Device array of maximum fitness values.
20 * @param genomes Device array of genomes.
21 * @param fitness Device array of fitness values.
22 */
23template<typename TFloat>
24__global__ void
25elite_population_replace_kernel(const uint32_t AGENTS,
26 const uint32_t DIMENSIONS,
27 const uint32_t* __restrict__ min_agent_idx,
28 const TFloat* __restrict__ max_agent_genome,
29 const TFloat* __restrict__ max_agent_fitness,
30 TFloat* __restrict__ genomes,
31 TFloat* __restrict__ fitness)
32{
33
34 const uint32_t i = threadIdx.x; // ISLE
35 const uint32_t ISLES = blockDim.x;
36
37 const uint32_t min_idx = min_agent_idx[i];
38
39 const uint32_t THREAD_OFFSET = ISLES * AGENTS;
40 const uint32_t BASE_IDX = min_idx + i * AGENTS;
41
42 TFloat* min_genome = genomes + BASE_IDX;
43
44 const TFloat* max_genome = max_agent_genome + i;
45
46 // Replace fitness & genome.
47 fitness[min_idx + i * AGENTS] = max_agent_fitness[i];
48 for (uint32_t k = 0; k < DIMENSIONS; k++) {
49 min_genome[k * THREAD_OFFSET] = max_genome[k * ISLES];
50 }
51}
